modal.js后面再研究
$(‘#btnId’).click(function(){
  creatModal(
'modalDialog',   //参数1：类型（modalDialog弹窗；messegeDialog提示框）
'modalId',           //参数2：弹窗id  //不要相同，这个窗口由creatModal创造
600,               //参数3：弹窗宽度
'mainContentId',     //参数4：主要内容id，自己写在页面上（如下）
'btnConfirmId',      //参数5：确认按钮id，已绑定方法fun
'弹窗title写在这里哦',  //参数6：弹窗title
'提示信息！',         //参数7：定义提示信息
fun                 //参数8：确定按钮方法
);
})

只有这两种情况
1, 'messegeDialog'，弹框id不同，弹框宽度，内容id为空，确认按钮为空（使用时其他人明显没有思考就写了），弹框title，提示信息自己写
2, 'modalDialog'， 弹框id不同，弹框宽度，内容id不同，确认按钮id不同，  			      		 弹框title，提示信息为空

creatModal('messegeDialog','messegeModalIdrecover',350,'','btnConfirmId','提示','提交设备恢复成功!'+data.msg);

creatModal('messegeDialog','messegeModalIdrecover',350,'','','提示','提交设备恢复成功!'+data.msg);
